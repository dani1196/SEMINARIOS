\begin{frame}{Modelo variacional \(\TV_q\)}
El problema principal en el caso discreto es:
\begin{equation}
 \min_{u\in\R^{|\Omega|}} f(u):=
\sum_{(i,j)\in\Omega}
\left(
\frac{\mu}{2}\abs{(\nabla u)_{ij}}^2
+\frac{\alpha}{q}\abs{(\nabla u)_{ij}}^q
+\frac{\lambda_{ij}}{2}\abs{(Ku-z)_{ij}}^2
\right).   
\label{2.1pv}
\end{equation}
\begin{block}{ParĂˇmetros}
    $\alpha>0$, $q\in (0,1)$, $0<\mu\ll \alpha$. 
\end{block}
\begin{columns}[T]
\column{0.3\textwidth}
\begin{block}{Fidelidad}
\[
\frac{\lambda_{ij}}{2}|(Ku-z)_{ij}|^2
\]
Obliga a que \(Ku\) se parezca a los datos observados.
\end{block}

\column{0.3\textwidth}
\begin{block}{TV\(_q\)}
\[
\frac{\alpha}{q}|(\nabla u)_{ij}|^q
\]
Promueve dispersiĂłn del gradiente y preserva bordes.
\end{block}

\column{0.3\textwidth}
\begin{block}{TĂ©rmino extra}
\[
\frac{\mu}{2}|(\nabla u)_{ij}|^2
\]
Aporta suavidad y estabilidad adicional.
\end{block}
\end{columns}
\end{frame}

% ============================================================
\begin{frame}{Existencia de soluciĂłn}
\begin{block}{Teorema 1}
    Supongamos que $\mu\geq 0$, $\alpha>0$,$q>0$, $\lambda_{ij}>0$ para todo $(i,j)\in\Omega$ y 
    $$
    \mathrm{Ker}\nabla \cap \mathrm{Ker} K=\{0\}.
    $$
    Entonces existe un minimizador global para el problema variacional \eqref{pv}.
\end{block}

\textbf{Idea de la DemostraciĂłn:}
    \begin{enumerate}
        \item Dado que $f$ es acotada inferiormente, basta probar que $f$ es coerciva (tiende a $+\infty$ cuando $\|u\|$ tiende a $+\infty$), puesto que si $f$ es coerciva y continua, entonces existe un minimizador. 
        \item La coercividad de $f$ se prueba por contradicciĂłn, de modo que se encuentra un elemento no nulo en la intersecciĂłn de $\mathrm{Ker}(\nabla)$ y $\mathrm{Ker}(K)$. Como esto no es posible de acuerdo a la hipĂłtesis, entonces $f$ es coerciva. 
    \end{enumerate}
\end{frame}

% ============================================================
\begin{frame}{Existencia de soluciĂłn}
 Para caracterizar la soluciĂłn Ăłptima $u$, se definen los conjuntos activo e inactivo de la siguiente forma:
    \begin{align*}
        \mathcal{A}(u):=\{(i,j)\in \Omega:|(\nabla u)_{ij}|\neq 0\},\quad
        \mathcal{I}(u):=\Omega \setminus \mathcal{A}(u).
    \end{align*}
  Como $f$ es no diferenciable en $\mathcal{I}(u)$, la ecuaciĂłn de Euler-Lagrange para caracterizar un punto estacionario se plantea como sigue 
    \begin{equation}
        \left\{\begin{array}{ll}
            -\mu\Delta u+K^T\lambda(Ku-z)+\alpha\nabla^T(|\nabla u|^{q-2}\nabla u)=0&\text{si}\;(i,j)\in\mathcal{A}(u)\\
            \nabla u=0&\text{si}\;(i,j)\in \mathcal{I}(u).
        \end{array}\right.
        \label{2.3}
    \end{equation}
    Dado que $f$ no es convexa, la soluciĂłn de \eqref{2.3} en general no es Ăşnica. 
\end{frame}

\begin{frame}{RegularizaciĂłn del tĂ©rmino $TV^q$}
    De este modo obtenemos el problema regularizado que aproxima al problema original, dado de la siguiente forma 
    \begin{equation}
    \min_{u\in\mathbb{R}^{|\Omega|}}f_{\gamma}(u):=\sum_{(i,j)\in \Omega}\left(\frac{\mu}{2}|(\nabla u)_{ij}|^2+\alpha\varphi_\gamma(|(\nabla u)_{ij}|)+\frac{\lambda_{ij}}{2}|(Ku-z)_{ij}|^2\right).
    \label{2.4}
    \end{equation}
    Dado que $f_\gamma$ es continuamente diferenciable, la ecuaciĂłn de Euler-Lagrange (condiciĂłn de primer orden del problema regularizado) estĂˇ dada por 
    \begin{equation}
     \nabla f_\gamma(u)=-\mu\Delta u+\mathbb{K}^T\lambda(Ku-z)+\alpha \nabla^T(\max(|\nabla u|,\gamma)^{q-2}\nabla u)
   \label{2.5} 
    \end{equation}
    \vspace{-0.4cm}
    \begin{alertblock}{Ventaja}
La funciĂłn \(\max(|\nabla u|,\gamma)\) evita dividir por cantidades cercanas a cero.
\end{alertblock}
\end{frame}

\begin{frame}{RegularizaciĂłn del tĂ©rmino $TV^q$}
    ÂżQuĂ© sucede si $\gamma\to 0^+$?
    \begin{block}{Teorema 2}
        Sean vĂˇlidos los supuestos del Teorema 1. AdemĂˇs supongamos que $(u^k)$ es una sucesiĂłn uniformemente acotada, donde cada $u^k$ es un punto estacionario del problema \eqref{2.4} que satisface \eqref{2.5}. Entonces cuando $\gamma\to 0^+$, existe una subsucesiĂłn de $(u^k)$ que converge a algĂşn $u^*\in \mathbb{R}^|\Omega|$, el cual satisface la ecuaciĂłn de Euler-Lagrange original \eqref{2.3}
    \end{block}
\end{frame}
% ============================================================
\begin{frame}{FormulaciĂłn primal-dual}
Reformulamos las condiciones de primer orden del problema regularizado \eqref{2.5}, que tambiĂ©n se muestran a continuaciĂłn
\begin{equation}
     \nabla f_\gamma(u)=-\mu\Delta u+\mathbb{K}^T\lambda(Ku-z)+\alpha \nabla^T(\max(|\nabla u|,\gamma)^{q-2}\nabla u),
    \end{equation}
introduciendo una variable dual $\overrightarrow{p}\in (\mathbb{R}^{|\Omega|})^2$, tal que
\begin{equation}
\begin{cases}
-\mu\Delta u + K^\top\lambda(Ku-z)+\alpha\nabla^\top \overrightarrow{p}=0,\\[4pt]
\max(|\nabla u|,\gamma)^{2-q}\overrightarrow{p}=\nabla u.
\end{cases}
\label{2.8}    
\end{equation}
\end{frame}

\begin{frame}{Newton semismooth primal-dual}

\end{frame}

\begin{frame}{ReponderaciĂłn de la ecuaciĂłn de Euler-Lagrange}
Dado \(u^k\), se define el peso como 
\[
    w^k=(\max(|\nabla u^k|,\gamma)^{q+r-2}, \qquad 0\leq r\leq 2-q.
\]

La ecuaciĂłn reponderada queda:
\[
-\mu\Delta u + K^\top\lambda(Ku-z)
+\alpha\nabla^\top\left(w^k\max(|\nabla u|,\gamma)^{-r}\nabla u\right)=0.
\]
Luego se introduce la variable dual reponderada 
$$
\overrightarrow{p}=w^k\max(|\nabla u|, \gamma)^{-r}\nabla u.
$$
Entonces \eqref{2.8} se escribe como 
\begin{equation}
\begin{cases}
-\mu\Delta u + K^\top\lambda(Ku-z)+\alpha\nabla^\top \overrightarrow{p}=0,\\[4pt]
(w^k)^{-1}\max(|\nabla u|,\gamma)^r\overrightarrow{p}=\nabla u.
\end{cases}
\label{3.2}    
\end{equation}
\end{frame}

\begin{frame}{Sistema lineal de Newton}
DespuĂ©s de aplicar una linealizaciĂłn generalizada a \eqref{3.2}, se obtiene un sistema de la forma:
\[
    \widetilde H^k(r)\,\delta u^{k+1}=-g^k,
\]
donde
\[
    g^k=\nabla f_\gamma(u^k).
\]

\begin{block}{Lectura computacional}
En cada iteraciĂłn se busca una direcciĂłn \(\delta u^{k+1}\) que reduzca el residuo de estacionariedad.
\end{block}
\end{frame}

\begin{frame}{El problema del Hessiano indefinido}
En problemas convexos, el Hessiano suele ser semidefinido positivo.

\vspace{0.2cm}
En este paper, ya que \(0<q<1\):
\begin{itemize}
    \item el funcional es no convexo,
    \item el Hessiano generalizado puede no ser positivo definido,
    \item una direcciĂłn de Newton puede no ser de descenso.
\end{itemize}

\end{frame}

\begin{frame}{RegularizaciĂłn \(R\)}
La reponderaciĂłn se interpreta como una regularizaciĂłn del Hessiano, denominada $R$-regularizaciĂłn, de modo que el sistema de Newton se escribe 
\[
    (H_k+\beta_k R_k)\delta u^{k+1}=-g^k.
\]

\begin{block}{Rol de \(\beta_k\)}
\begin{itemize}
    \item Si \(\beta_k\) es grande, el sistema es mĂˇs estable.
    \item Si \(\beta_k\to 0\), se recupera el comportamiento de Newton puro.
\end{itemize}
\end{block}

\end{frame}

\begin{frame}{RegiĂłn de confianza}
La versiĂłn $R$-regularizada del mĂ©todo de Newton generalizado con $\beta$ fijo es linealmente convergente. Sin embargo, se propone una nueva versiĂłn adaptativamente $R$-regularizada del mĂ©todo de Newton generalizado que alcanza convergencia local superlineal. Para ello se requiere una estrategia de actualizaciĂłn adecuada para $\beta>0$. Por lo que se propone un esquema de tipo regiĂłn de confianza. Dado un $u^k$, se modela a $f_\gamma$ localmente mediante una funciĂłn cuadrĂˇtica $h:\mathbb{R}^|\Omega|\to\mathbb{R}$ con
\[
    h^k(d)=f_\gamma(u^k)+(g^k)^\top d+\frac12 d^\top H_+^kd.
\]

La direcciĂłn se controla mediante una restricciĂłn tipo regiĂłn de confianza:
\[
    \frac12 d^\top R_{k,\varepsilon}^+d\leq \frac12\sigma_k^2.
\]
\end{frame}

\begin{frame}{Algoritmo 3.6: MĂ©todo de Newton adaptativamente regularizado}
    % Reducimos la fuente al mĂ­nimo legible para proyecciĂłn e ignoramos espaciados verticales extras
    \tiny
    \setlength{\leftmargini}{0.5cm}
    \setlength{\topsep}{0pt}
    \setlength{\partopsep}{0pt}
    \setlength{\parsep}{0pt}
    \setlength{\itemsep}{0pt}
    
    \begin{algorithmic}[1]
        \Require input parameters $1 - q \le \beta_{\max} \le 2 - q$, $c > 0$, $0 < \rho_1 \le \rho_2 < 1$, $0 < \kappa_1 < 1 < \kappa_2$, $0 < \varepsilon \ll \alpha$, $0 < \epsilon_d \le \bar{\epsilon}_d$, $0 < \tau_1 < 1/2$, $\tau_1 < \tau_2 < 1$.
        \State Initialize $(u^0, \vec{p}^0)$, the regularization scalar $\beta^0 \ge 0$, and the trust-region radius $\sigma^0 > 0$. Set $k := 0$.
        \Repeat \text{ \{outer loop\}}
            \State Generate $H^k_+$, $R^k_{+,\varepsilon}$, and $g^k$.
            \Repeat \text{ \{inner loop\}}
                \State Solve $(H^k_+ + \beta^k R^k_{+,\varepsilon})d^k = -g^k$ for $d^k$.
                \If{$-(g^k)^\top d^k / (\|g^k\| \|d^k\|) < \epsilon_d$} \ Set $\beta^k := \beta_{\max}$ and return to step 5. \EndIf
                \If{$\beta^k = \beta_{\max}$ \textbf{and} $(d^k)^\top R^k_{+,\varepsilon}d^k > (\sigma^k)^2$}
                    \State Set $\sigma^k := \sqrt{(d^k)^\top R^k_{+,\varepsilon}d^k}$ and go to step 15.
                \EndIf
                \State Update $\beta^k := \beta^k + ((d^k)^\top R^k_{+,\varepsilon}d^k - (\sigma^k)^2)/(2c)$.
                \State Project $\beta^k$ onto $[0, \beta_{\max}]$; i.e., set $\beta^k := \max(\min(\beta^k, \beta_{\max}), 0)$.
            \Until{the stopping criterion for the inner loop is fulfilled.}
            \State Evaluate $\rho^k := [f_\gamma(u^k) - f_\gamma(u^k + d^k)] / [f_\gamma(u^k) - (f_\gamma(u^k) + (g^k)^\top d^k + \frac{1}{2}(d^k)^\top H^k_+ d^k)]$.
            \If{$\rho^k < \rho_1$} \ Set $\sigma^{k+1} := \kappa_1 \sigma^k$.
            \ElsIf{$\rho^k > \rho_2$} \ Set $\sigma^{k+1} := \kappa_2 \sigma^k$.
            \Else \ $\sigma^{k+1} := \sigma^k$.
            \EndIf
            \State Determine step size $a^k$ along $d^k$ such that $u^{k+1} = u^k + a^k d^k$ satisfies the Wolfeâ€“Powell conditions:
            \Statex \hspace{\algorithmicindent} (3.35) $f_\gamma(u^{k+1}) \le f_\gamma(u^k) + \tau_1 a^k \nabla f_\gamma(u^k)^\top d^k$ \ \textbf{and} \ (3.36) $\nabla f_\gamma(u^{k+1})^\top d^k \ge \tau_2 \nabla f_\gamma(u^k)^\top d^k$.
            \State Set $\delta u^{k+1} := d^k$ and compute $\delta\vec{p}^{k+1}$ via (3.9). Update $u^{k+1} := u^k + a^k \delta u^{k+1}$ and $\vec{p}^{k+1} := \vec{p}^k + a^k \delta\vec{p}^{k+1}$.
            \State Set $\beta^{k+1} := \beta^k$ and $k := k + 1$.
        \Until{the stopping criterion for the outer loop is fulfilled.}
    \end{algorithmic}
\end{frame}
% ============================================================
\begin{frame}{Convergencia global}
\begin{block}{Resultado}
Bajo las condiciones del algoritmo y usando bĂşsqueda lineal de Wolfe--Powell,
\[
    \lim_{k\to\infty}\nabla \|f_\gamma(u^k)\|=0.
\]
\end{block}

\begin{alertblock}{InterpretaciĂłn}
El algoritmo no necesariamente encuentra un mĂ­nimo global, porque el problema es no convexo, pero sĂ­ converge hacia un punto estacionario del problema Huberizado.
\end{alertblock}
\end{frame}

% ============================================================
\begin{frame}{Convergencia local superlineal}
\begin{block}{Resultado local}
Si la sucesiĂłn converge a un punto estacionario adecuado y los elementos del B-Hessiano son definidos positivos, entonces:
\[
    \norm{u^{k+1}-u^*}=o\left(\norm{u^k-u^*}\right).
\]
\end{block}

\end{frame}
