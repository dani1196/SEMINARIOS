\begin{enumerate}
    \item Expandir la capacidad de los algoritmos para recuperar imágenes usando la norma $\ell_q$.

\begin{frame}{Notación básica}
\begin{itemize}
    \item \(\Omega\): dominio discreto de píxeles.
    \item \(u\in\R^{|\Omega|}\): imagen reconstruida.
    \item \(z\in\R^{|\Omega|}\): datos observados.
    \item \(K\in\R^{|\Omega|\times |\Omega|}\): operador de degradación.
    \item \(\nabla u = (\nabla_x u,\nabla_y u)\): gradiente discreto.
    \item $\Delta=-\nabla_x^T\nabla_x-\nabla_y^T\nabla_y:$ Laplaciano discreto.
    \item \(\lambda_{ij}>0\): coeficiente de fidelidad.
\end{itemize}

\begin{block}{Para cada píxel}
\[
    \abs{(\nabla u)_{ij}} = \sqrt{((\nabla_xu)_{ij})^2+((\nabla_yu)_{ij})^2}.
\]
\end{block}
\end{frame}

\begin{frame}{Modelo variacional \(\TV_q\)}
El problema es:
\begin{equation}
 \min_{u\in\R^{|\Omega|}} f(u):=
\sum_{(i,j)\in\Omega}
\left(
\frac{\mu}{2}\abs{(\nabla u)_{ij}}^2
+\frac{\alpha}{q}\abs{(\nabla u)_{ij}}^q
+\frac{\lambda_{ij}}{2}\abs{(Ku-z)_{ij}}^2
\right).   
\label{2.1pv}
\end{equation}
\begin{block}{Parámetros}
    $\alpha>0$, $q\in (0,1)$, $0<\mu\ll \alpha$. 
\end{block}
\begin{columns}[T]
\column{0.3\textwidth}
\begin{block}{Fidelidad}
\[
\frac{\lambda_{ij}}{2}|(Ku-z)_{ij}|^2
\]
Obliga a que \(Ku\) se parezca a los datos observados.
\end{block}

\column{0.3\textwidth}
\begin{block}{TV\(_q\)}
\[
\frac{\alpha}{q}|(\nabla u)_{ij}|^q
\]
Promueve dispersión del gradiente y preserva bordes.
\end{block}

\column{0.3\textwidth}
\begin{block}{Término extra}
\[
\frac{\mu}{2}|(\nabla u)_{ij}|^2
\]
Aporta suavidad y estabilidad adicional.
\end{block}
\end{columns}
\end{frame}

% ============================================================
\begin{frame}{Existencia de solución}
\begin{block}{Teorema 1}
    Supongamos que $\mu\geq 0$, $\alpha>0$,$q>0$, $\lambda_{ij}>0$ para todo $(i,j)\in\Omega$ y 
    $$
    \mathrm{Ker}\nabla \cap \mathrm{Ker} K=\{0\}.
    $$
    Entonces existe un minimizador global para el problema variacional \eqref{2.1pv}.
\end{block}

\textbf{Idea de la Demostración:}
    \begin{enumerate}
        \item Dado que $f$ es acotada inferiormente, basta probar que $f$ es coerciva (tiende a $+\infty$ cuando $\|u\|$ tiende a $+\infty$), puesto que si $f$ es coerciva y continua, entonces existe un minimizador. 
        \item La coercividad de $f$ se prueba por contradicción, de modo que se encuentra un elemento no nulo en la intersección de $\mathrm{Ker}(\nabla)$ y $\mathrm{Ker}(K)$. Como esto no es posible de acuerdo a la hipótesis, entonces $f$ es coerciva. 
    \end{enumerate}
\end{frame}

% ============================================================
\begin{frame}{Existencia de solución}
 Para caracterizar la solución óptima $u$, se definen los conjuntos activo e inactivo de la siguiente forma:
    \begin{align*}
        \mathcal{A}(u):=\{(i,j)\in \Omega:|(\nabla u)_{ij}|\neq 0\},\quad
        \mathcal{I}(u):=\Omega \setminus \mathcal{A}(u).
    \end{align*}
  Como $f$ es no diferenciable en $\mathcal{I}(u)$, la ecuación de Euler-Lagrange para caracterizar un punto estacionario se plantea como sigue 
    \begin{equation}
        \left\{\begin{array}{ll}
            -\mu\Delta u+K^T\lambda(Ku-z)+\alpha\nabla^T(|\nabla u|^{q-2}\nabla u)=0&\text{si}\;(i,j)\in\mathcal{A}(u)\\
            \nabla u=0&\text{si}\;(i,j)\in \mathcal{I}(u).
        \end{array}\right.
        \label{2.3}
    \end{equation}
    Dado que $f$ no es convexa, la solución de \eqref{2.3} en general no es única. 
\end{frame}

% ============================================================
\begin{frame}{Regularización del término $TVq$}
  \begin{columns}[c] % El parámetro [c] centra verticalmente el contenido de las columnas
        
        % Primera columna: Texto y Ecuación
        \begin{column}{0.5\textwidth}
            Regularización mediante una función de Huber $\varphi_\gamma$:
            \begin{equation*}
                \varphi_\gamma(s) = 
                \begin{cases} 
                    \frac{1}{q} |s|^q - \left(\frac{1}{q} - \frac{1}{2}\right)\gamma^q & \text{si } |s| > \gamma, \\ 
                    \frac{1}{2} \gamma^{q-2} |s|^2 & \text{si } |s| \le \gamma, 
                \end{cases}
            \end{equation*}
            con $\gamma>0$ pequeño.
        \end{column}
        
        % Segunda columna: Imagen

\end{frame}
% ============================================================
\begin{frame}{Regularización del término $TVq$}
    De este modo obtenemos el problema regularizado que aproxima al problema original, dado de la siguiente forma 
    \begin{equation}
    \min_{u\in\mathbb{R}^{|\Omega|}}f_{\gamma}(u):=\sum_{(i,j)\in \Omega}\left(\frac{\mu}{2}|(\nabla u)_{ij}|^2+\alpha\varphi_\gamma(|(\nabla u)_{ij}|)+\frac{\lambda_{ij}}{2}|(Ku-z)_{ij}|^2\right).
    \label{2.4}
    \end{equation}
    Dado que $f_\gamma$ es continuamente diferenciable, la ecuación de Euler-Lagrange (condición de primer orden del problema regularizado) está dada por 
    \begin{equation}
     \nabla f_\gamma(u)=-\mu\Delta u+\mathbb{K}^T\lambda(Ku-z)+\alpha \nabla^T(\max(|\nabla u|,\gamma)^{q-2}\nabla u)=0.
   \label{2.5} 
    \end{equation}
    \vspace{-0.4cm}
    \begin{alertblock}{Ventaja}
La función \(\max(|\nabla u|,\gamma)\) evita dividir por cantidades cercanas a cero.
\end{alertblock}
\end{frame}
% ============================================================
\begin{frame}{Regularización del término $TVq$}
    ¿Qué sucede si $\gamma\to 0^+$?
    \begin{block}{Teorema 2}
        Sean válidos los supuestos del Teorema 1. Además supongamos que $(u^k)$ es una sucesión uniformemente acotada, donde cada $u^k$ es un punto estacionario del problema \eqref{2.4} que satisface \eqref{2.5}. Entonces cuando $\gamma\to 0^+$, existe una subsucesión de $(u^k)$ que converge a algún $u^*\in \mathbb{R}^|\Omega|$, el cual satisface la ecuación de Euler-Lagrange original \eqref{2.3}.
    \end{block}
\end{frame}
% ============================================================
\begin{frame}{Formulación primal-dual}
Reformulamos las condiciones de primer orden del problema regularizado \eqref{2.5}, que también se muestran a continuación
\begin{equation}
     \nabla f_\gamma(u)=-\mu\Delta u+\mathbb{K}^T\lambda(Ku-z)+\alpha \nabla^T(\max(|\nabla u|,\gamma)^{q-2}\nabla u),
    \end{equation}
introduciendo una variable dual $\overrightarrow{p}\in (\mathbb{R}^{|\Omega|})^2$, tal que
\begin{equation}
\begin{cases}
-\mu\Delta u + K^\top\lambda(Ku-z)+\alpha\nabla^\top \overrightarrow{p}=0,\\[4pt]
\max(|\nabla u|,\gamma)^{2-q}\overrightarrow{p}=\nabla u.
\end{cases}
\label{2.8}    
\end{equation}
\end{frame}
% ============================================================
\begin{frame}{Newton semismooth primal-dual}
El objetivo es resolver el sistema no lineal primal-dual mediante un método de Newton generalizado.

\begin{block}{Motivación}
Los métodos Newton pueden ser muy rápidos localmente, pero aquí aparecen dos dificultades:
\begin{itemize}
    \item no convexidad del funcional,
    \item Hessiano generalizado indefinido.
\end{itemize}
\end{block}

\begin{alertblock}{Estrategia del paper}
Combinar Newton semismooth con repoderación y regularización tipo región de confianza.
\end{alertblock}
\end{frame}

\begin{frame}{Reponderación de la ecuación de Euler-Lagrange}
Dado \(u^k\), se define el peso como 
\[
    w^k=(\max(|\nabla u^k|,\gamma)^{q+r-2}, \qquad 0\leq r\leq 2-q,
\]

con $r$ el parámetro de ponderación. La ecuación reponderada queda:
\[
-\mu\Delta u + K^\top\lambda(Ku-z)
+\alpha\nabla^\top\left(w^k\max(|\nabla u|,\gamma)^{-r}\nabla u\right)=0.
\]
Luego se introduce la variable dual reponderada 
$$
\overrightarrow{p}=w^k\max(|\nabla u|, \gamma)^{-r}\nabla u.
$$
Entonces \eqref{2.8} se escribe como 
\begin{equation}
\begin{cases}
-\mu\Delta u + K^\top\lambda(Ku-z)+\alpha\nabla^\top \overrightarrow{p}=0,\\[4pt]
(w^k)^{-1}\max(|\nabla u|,\gamma)^r\overrightarrow{p}=\nabla u.
\end{cases}
\label{3.2}    
\end{equation}
\end{frame}

\begin{frame}{Sistema lineal de Newton}
Después de aplicar una linealización generalizada a \eqref{3.2}, se obtiene un sistema de la forma:
\[
    \widetilde H^k(r)\,\delta u^{k+1}=-g^k,
\]
donde
\[
    g^k=\nabla f_\gamma(u^k).
\]

\begin{block}{Lectura computacional}
En cada iteración se busca una dirección \(\delta u^{k+1}\) que reduzca el residuo de estacionariedad.
\end{block}
\end{frame}

% ============================================================
\begin{frame}{El problema del Hessiano indefinido}
En problemas convexos, el Hessiano suele ser semidefinido positivo.

\vspace{0.2cm}
En este paper, ya que \(0<q<1\):
\begin{itemize}
    \item el funcional es no convexo,
    \item el Hessiano generalizado puede no ser positivo definido,
    \item una dirección de Newton puede no ser de descenso.
\end{itemize}

\end{frame}

% 23
% ============================================================
\begin{frame}{Regularización \(R\)}
La reponderación se interpreta como una regularización del Hessiano, denominada $R$-regularización, de modo que el sistema de Newton se escribe 
\[
    (H^k+\beta^k R^k)\delta u^{k+1}=-g^k,
\]
donde $H^k:=\tilde{H}^k(2-q)$ es el B-Hessiano en el método de Newton primal-dual sin reponderación, $\beta=\beta^k=2-q-r$.

\begin{block}{Rol de \(\beta_k\)}
\begin{itemize}
    \item Si \(\beta_k\) es grande, el sistema es más estable.
    \item Si \(\beta_k\to 0\), se recupera el comportamiento de Newton puro.
\end{itemize}
\end{block}

\end{frame}

\begin{frame}{Región de confianza}
La versión $R$-regularizada del método de Newton generalizado con $\beta$ fijo es linealmente convergente. \vspace{0.15cm}

Sin embargo, se propone una nueva versión adaptativamente $R$-regularizada del método de Newton generalizado que alcanza convergencia local superlineal. \vspace{0.15cm}

Para ello se requiere una estrategia de actualización adecuada para $\beta>0$. Por lo que se propone un esquema de tipo región de confianza. Dado un $u^k$, se modela a $f_\gamma$ localmente mediante una función cuadrática $h:\mathbb{R}^{|\Omega|}\to\mathbb{R}$ con
\[
    h^k(d)=f_\gamma(u^k)+(g^k)^\top d+\frac12 d^\top H_+^kd.
\]

La dirección $d$ es tal que 
\[
    \frac12 d^\top R_{+,\epsilon}^kd\leq \frac12\sigma_k^2,
\]
donde $R_+$ es una versión simetrizada del operador de regularización $R^k$, $R_{+,\epsilon}^k:=R_+^k+\epsilon I$, con un parámetro arbitrario de regularización fijo $0<\epsilon\ll \alpha$ y $\sigma^k>0$ representa el radio de la región de confianza. 
\end{frame}

% ============================================================
\begin{frame}{Algoritmo: Método de Newton adaptativamente regularizado}
    % Reducimos la fuente al mínimo legible para proyección e ignoramos espaciados verticales extras
    \tiny
    \setlength{\leftmargini}{0.5cm}
    \setlength{\topsep}{0pt}
    \setlength{\partopsep}{0pt}
    \setlength{\parsep}{0pt}
    \setlength{\itemsep}{0pt}
    
    \begin{algorithmic}[1]
        \Require input parameters $1 - q \le \beta_{\max} \le 2 - q$, $c > 0$, $0 < \rho_1 \le \rho_2 < 1$, $0 < \kappa_1 < 1 < \kappa_2$, $0 < \varepsilon \ll \alpha$, $0 < \epsilon_d \le \bar{\epsilon}_d$, $0 < \tau_1 < 1/2$, $\tau_1 < \tau_2 < 1$.
        \State Initialize $(u^0, \vec{p}^0)$, the regularization scalar $\beta^0 \ge 0$, and the trust-region radius $\sigma^0 > 0$. Set $k := 0$.
        \Repeat \text{ \{outer loop\}}
            \State Generate $H^k_+$, $R^k_{+,\varepsilon}$, and $g^k$.
            \Repeat \text{ \{inner loop\}}
                \State Solve $(H^k_+ + \beta^k R^k_{+,\varepsilon})d^k = -g^k$ for $d^k$.
                \If{$-(g^k)^\top d^k / (\|g^k\| \|d^k\|) < \epsilon_d$} \
                \State Set $\beta^k := \beta_{\max}$ and return to step 5. \EndIf
                \If{$\beta^k = \beta_{\max}$ \textbf{and} $(d^k)^\top R^k_{+,\varepsilon}d^k > (\sigma^k)^2$}
                    \State Set $\sigma^k := \sqrt{(d^k)^\top R^k_{+,\varepsilon}d^k}$ and go to step 15.
                \EndIf
                \State Update $\beta^k := \beta^k + ((d^k)^\top R^k_{+,\varepsilon}d^k - (\sigma^k)^2)/(2c)$.
                \State Project $\beta^k$ onto $[0, \beta_{\max}]$; i.e., set $\beta^k := \max(\min(\beta^k, \beta_{\max}), 0)$.
            \Until{the stopping criterion for the inner loop is fulfilled.}
            \State Evaluate $\rho^k := [f_\gamma(u^k) - f_\gamma(u^k + d^k)] / [f_\gamma(u^k) - (f_\gamma(u^k) + (g^k)^\top d^k + \frac{1}{2}(d^k)^\top H^k_+ d^k)]$.
            \If{$\rho^k < \rho_1$} \ Set $\sigma^{k+1} := \kappa_1 \sigma^k$.
            \ElsIf{$\rho^k > \rho_2$} \ Set $\sigma^{k+1} := \kappa_2 \sigma^k$.
            \Else \ $\sigma^{k+1} := \sigma^k$.
            \EndIf
            \State Determine step size $a^k$ along $d^k$ such that $u^{k+1} = u^k + a^k d^k$ satisfies the Wolfe–Powell conditions:
            \Statex \hspace{\algorithmicindent} (WP1) $f_\gamma(u^{k+1}) \le f_\gamma(u^k) + \tau_1 a^k \nabla f_\gamma(u^k)^\top d^k$ \ \textbf{and} \ (WP2) $\nabla f_\gamma(u^{k+1})^\top d^k \ge \tau_2 \nabla f_\gamma(u^k)^\top d^k$.
            \State Set $\delta u^{k+1} := d^k$ and compute $\delta\vec{p}^{k+1}$ via (3.9). Update $u^{k+1} := u^k + a^k \delta u^{k+1}$ and $\vec{p}^{k+1} := \vec{p}^k + a^k \delta\vec{p}^{k+1}$.
            \State Set $\beta^{k+1} := \beta^k$ and $k := k + 1$.
        \Until{the stopping criterion for the outer loop is fulfilled.}
    \end{algorithmic}
\end{frame}
% ============================================================
\begin{frame}{Convergencia global}
\begin{block}{Teorema (Convergencia global)}
Sea $u^{k+1}=u^k+a^kd^k$ tal que las condiciones de Wolfe-Powell (WP1) y (WP2) se satisfacen. Entonces 
\[
    \lim_{k\to\infty}\|\nabla f_\gamma(u^k)\|=0.
\]
\end{block}

\begin{alertblock}{Interpretación}
El algoritmo no necesariamente encuentra un mínimo global, porque el problema es no convexo, pero sí converge hacia un punto estacionario del problema Huberizado.
\end{alertblock}
\end{frame}

% ============================================================
\begin{frame}{Convergencia local superlineal}
\begin{block}{Teorema (Convergencia local)}
Sea $(d^k)$ generada por el Algoritmo, y sea la sucesión $(u^k,\vec{p}^k)$ convergente a algún $(u^*,\vec{p}^*)$ que satisface el sistema de Euler-Lagrange \eqref{2.8}. Supongamos que todos los elementos en $\partial_{B}^2f_\gamma(u^*)$ son estrictamente definidos positivos. Entonces el Algoritmo es localmente superlinealmente convergente, es decir, para $k$ suficientemente grande tenemos que 
$$
    \norm{u^{k+1}-u^*}=o\left(\norm{u^k-u^*}\right)\qquad\text{para}\;k\to\infty.
$$
\end{block}

\begin{block}{Notación}
$\partial_B^2 f_\gamma(u)$ es el conjunto formado por todas las matrices en $\mathbb{R}^{|\Omega|\times|\Omega|}$ que son límites de sucesiones de la forma $\nabla^2 f_\gamma(u^k)$ con $u^k\to u$ y $\nabla f_\gamma$ diferenciable en $u^k$, es decir, 
   $$ \partial_B^2f_\gamma(u):=\{\lim \nabla^2f_\gamma(u^k):u^k\to u, \nabla f_\gamma\;\text{es diferenciable en}\;u^k\}.$$

Además, $\nabla_B^2f_\gamma(u)$ corresponde al B-Hessiano de $f_\gamma$ en $u$. 
\end{block}
\end{frame}

\begin{frame}{Resumen de resultados}
\begin{enumerate}
    \item Introduce un modelo \(\TV_q\) no convexo para restauración de imágenes.
    \item Prueba existencia de minimizador en el caso discreto.
    \item Regulariza localmente el término no diferenciable mediante una función tipo Huber.
    \item Propone un método Newton primal-dual semismooth.
    \item Usa regularización basada en región de confianza para tratar Hessianos indefinidos.
    \item Demuestra convergencia global y convergencia local superlineal hacia puntos estacionarios del problema regularizado.
\end{enumerate}
\vspace{1cm}
{\footnotesize
M. Hintermüller y T. Wu (2013). \emph{Nonconvex TVq-Models in Image Restoration: Analysis and a Trust-Region Regularization-Based Superlinearly Convergent Solver}. SIAM Journal on Imaging Sciences, 6(3), 1385--1415.
}
